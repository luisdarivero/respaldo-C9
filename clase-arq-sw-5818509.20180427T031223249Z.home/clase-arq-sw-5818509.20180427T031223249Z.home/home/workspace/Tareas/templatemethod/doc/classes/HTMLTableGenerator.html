def generate
  #[ "a", "b", "c" ].join #=> "abc"
  cadena = "<table>\n#{generate_header_row + (@data.map {|x| generate_row(x)}).join}</table>\n"
  #puts("resultado-----\n" + cadena + "\ntermina-------------\n")
  return cadena
