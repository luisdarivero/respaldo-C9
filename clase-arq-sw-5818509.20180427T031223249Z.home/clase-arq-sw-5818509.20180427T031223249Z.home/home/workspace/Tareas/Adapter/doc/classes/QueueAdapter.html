def peek
    if @q.empty?
      nil
    else
      temp = []
      while !@q.empty?
          temp<<@q.remove
      end
      
      
      for element in temp
          @q.insert(element)
      end
      return temp[0]
    end
end

def push(x)
    temp = []
    temp << x
    while !@q.empty?
        temp<<@q.remove
    end

        @q.insert(element)
    end
